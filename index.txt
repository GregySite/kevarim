(function(){
  let startY=0,pulling=false,triggered=false;
  const THRESHOLD=72;
  const ptr=document.getElementById('ptr');
  const icon=document.getElementById('ptr-icon');
  const txt=document.getElementById('ptr-txt');

  function refresh(){
    triggered=true;
    icon.textContent='↻';
    icon.style.animation='spin .8s linear infinite';
    txt.setAttribute('data-i18n','searching');
    txt.textContent=t('searching');
    // Force fresh fetch from Overpass (ignore KEVARIM_DATA & cache)
    const seen=new Set();
    const r=S.rangeKm*1000;
    if(!S.lat||!S.lng){reset();return;}
    async function go(){
      async function overpass(q){const res=await fetch('https://overpass-api.de/api/interpreter',{method:'POST',body:q});if(!res.ok)throw new Error(`HTTP ${res.status}`);const d=await res.json();return d;}
      const qJ=`[out:json][timeout:30];(node["landuse"="cemetery"]["religion"="jewish"](around:${r},${S.lat},${S.lng});way["landuse"="cemetery"]["religion"="jewish"](around:${r},${S.lat},${S.lng});node["amenity"="grave_yard"]["religion"="jewish"](around:${r},${S.lat},${S.lng});way["amenity"="grave_yard"]["religion"="jewish"](around:${r},${S.lat},${S.lng}););out center;`;
      const qA=`[out:json][timeout:30];(node["landuse"="cemetery"](around:${r},${S.lat},${S.lng});way["landuse"="cemetery"](around:${r},${S.lat},${S.lng});node["amenity"="grave_yard"](around:${r},${S.lat},${S.lng});way["amenity"="grave_yard"](around:${r},${S.lat},${S.lng}););out center;`;
      try{
        let d=await overpass(qJ);
        if(!d.elements.length) d=await overpass(qA);
        const fresh=processElements(d.elements,false,seen);
        S.cems=fresh;S.dataOK=true;dot('d-dat','ok');
        // Phase 2 graves
        try{
          const qG=`[out:json][timeout:20];(node["historic"="tomb"]["name"](around:${r},${S.lat},${S.lng});node["historic"="grave"]["name"](around:${r},${S.lat},${S.lng}););out;`;
          const d2=await overpass(qG);
          const graves=processElements(d2.elements,true,seen);
          if(graves.length) S.cems=[...S.cems,...graves];
        }catch(e){}
        document.getElementById('cem-count').textContent=filteredCems().length;
        document.getElementById('cache-icon').textContent='☁';
        clearPins();
      }catch(e){dot('d-dat','err');}
      reset();
    }
    go();
  }

  function reset(){
    triggered=false;pulling=false;
    ptr.style.height='0';
    icon.style.animation='';
    icon.textContent='↓';
    setTimeout(()=>ptr.style.height='0',300);
  }

  document.addEventListener('touchstart',e=>{
    if(document.getElementById('list-view').classList.contains('open')) return;
    if(document.getElementById('detail').classList.contains('open')) return;
    startY=e.touches[0].clientY;pulling=false;triggered=false;
  },{passive:true});

  document.addEventListener('touchmove',e=>{
    if(triggered) return;
    if(document.getElementById('list-view').classList.contains('open')) return;
    if(document.getElementById('detail').classList.contains('open')) return;
    const dy=e.touches[0].clientY-startY;
    if(dy<10) return;
    pulling=true;
    const h=Math.min(dy*0.45,THRESHOLD+10);
    ptr.style.height=h+'px';
    const ratio=Math.min(dy/THRESHOLD,1);
    icon.style.transform=`rotate(${ratio*180}deg)`;
    icon.style.animation='';
    icon.textContent=dy>THRESHOLD?'↺':'↓';
    txt.textContent=dy>THRESHOLD?t('searching'):'';
  },{passive:true});

  document.addEventListener('touchend',e=>{
    if(!pulling||triggered) return;
    const dy=e.changedTouches[0].clientY-startY;
    if(dy>THRESHOLD) refresh();
    else reset();
  },{passive:true});

  // Spin keyframe
  const st=document.createElement('style');
  st.textContent='@keyframes spin{to{transform:rotate(360deg)}}';
  document.head.appendChild(st);
})();

function haversine(la1,lo1,la2,lo2){
  const R=6371,d2r=Math.PI/180;
  const dLa=(la2-la1)*d2r,dLo=(lo2-lo1)*d2r;
  const a=Math.sin(dLa/2)**2+Math.cos(la1*d2r)*Math.cos(la2*d2r)*Math.sin(dLo/2)**2;
  return R*2*Math.atan2(Math.sqrt(a),Math.sqrt(1-a));
}
function bearing(la1,lo1,la2,lo2){
  const d2r=Math.PI/180;
  const y=Math.sin((lo2-lo1)*d2r)*Math.cos(la2*d2r);
  const x=Math.cos(la1*d2r)*Math.sin(la2*d2r)-Math.sin(la1*d2r)*Math.cos(la2*d2r)*Math.cos((lo2-lo1)*d2r);
  return(Math.atan2(y,x)*180/Math.PI+360)%360;
}

  if(km<5) return`${km.toFixed(1)} km`;
  if(km<20) return`${Math.round(km)} km`;
  return`${Math.round(km/5)*5} km`;
}
function smoothAlpha(km){return Math.max(0.025,0.3/(1+km*1.3));}

const CACHE_KEY='kevarim_v1',CACHE_TTL=30*24*60*60*1000;
function cacheKey(la,lo,km){return`${(Math.round(la/0.05)*0.05).toFixed(2)},${(Math.round(lo/0.05)*0.05).toFixed(2)},${km}`;}
function cacheLoad(la,lo,km){
  try{const s=JSON.parse(localStorage.getItem(CACHE_KEY)||'{}');const e=s[cacheKey(la,lo,km)];return e&&Date.now()-e.ts<CACHE_TTL?e.cems:null;}catch(e){return null;}
}
function cacheSave(la,lo,km,cems){
  try{const s=JSON.parse(localStorage.getItem(CACHE_KEY)||'{}');const keys=Object.keys(s);if(keys.length>=20)delete s[keys.sort((a,b)=>s[a].ts-s[b].ts)[0]];s[cacheKey(la,lo,km)]={ts:Date.now(),cems:cems.map(c=>({lat:c.lat,lng:c.lng,name:c.name,name2:c.name2||'',sublabel:c.sublabel,isGrave:c.isGrave}))};localStorage.setItem(CACHE_KEY,JSON.stringify(s));}catch(e){}
}

function pickName(tags){
  return(lang==='he'&&(tags['name:he']||tags.name))||(lang==='fr'&&(tags['name:fr']||tags.name))||(lang==='en'&&(tags['name:en']||tags.name))||tags.name||null;
}
function pickName2(tags,primary){
  // Secondary name in a different script from primary
  const alts=[tags['name:en'],tags['name:fr'],tags['name:he']].filter(n=>n&&n!==primary&&n.trim());
  return alts[0]||'';
}
function processElements(elements,isGrave,seen){
  return elements.map(el=>{
    const la=el.lat||el.center?.lat,lo=el.lon||el.center?.lon;
    if(!la||!lo) return null;
    const key=`${la.toFixed(4)},${lo.toFixed(4)}`;
    if(seen.has(key)) return null;
    seen.add(key);
    const tags=el.tags||{};
    const name=pickName(tags);
    if(!name) return null;
    const name2=pickName2(tags,name);
    const sublabel=isGrave?'✡ '+t('grave'):'✡ '+t('cemetery');
    return{lat:la,lng:lo,name,name2,sublabel,isGrave};
  }).filter(Boolean);
}

function clearPins(){document.getElementById('ar-overlay').innerHTML='';pinEls.clear();}

function render(){
  const overlay=document.getElementById('ar-overlay');
  if(!S.gpsOK||!S.dataOK){clearPins();return;}
  const W=window.innerWidth,H=window.innerHeight;
  const isHe=lang==='he';
  const FULL_H=130,GAP=6,GROUND_Y=H*0.72;
  const cems=filteredCems();

  const visible=cems.map(c=>{
    const dist=haversine(S.lat,S.lng,c.lat,c.lng);
    const bear=bearing(S.lat,S.lng,c.lat,c.lng);
    const rel=angDiff(bear,S.heading);
    return{...c,dist,bear,rel};
  }).filter(c=>Math.abs(c.rel)<=S.FOV/2+5);

  for(const c of visible){
    c._id=`${c.lat.toFixed(4)},${c.lng.toFixed(4)}`;
    c._bx=W/2+(c.rel/(S.FOV/2))*(W/2);
    c._op=Math.max(0.55,1-Math.min(c.dist/S.rangeKm,1)*0.32);
  }

  // Group overlapping pins
  const byX=[...visible].sort((a,b)=>a._bx-b._bx);
  const groups=[];
  if(byX.length){
    let g=[byX[0]];
    for(let i=1;i<byX.length;i++){if(byX[i]._bx-byX[i-1]._bx<200+GAP)g.push(byX[i]);else{groups.push(g);g=[byX[i]];}}
    groups.push(g);
  }
  for(const grp of groups){
    grp.sort((a,b)=>b.dist-a.dist);
    const colX=grp.reduce((s,c)=>s+c._bx,0)/grp.length;
    grp.forEach((c,idx)=>{c._colX=colX;c._targetOff=(grp.length-1-idx)*(FULL_H+GAP);c._isBottom=idx===grp.length-1;});
  }
  for(const c of visible){
    const prevX=smoothedPos.has(c._id)?smoothedPos.get(c._id):c._colX;
    smoothedPos.set(c._id,prevX+smoothAlpha(c.dist)*(c._colX-prevX));
    c._x=smoothedPos.get(c._id);
    const prevOff=smoothedStem.has(c._id)?smoothedStem.get(c._id):c._targetOff;
    smoothedStem.set(c._id,prevOff+(c._targetOff>prevOff?0.18:0.09)*(c._targetOff-prevOff));
    c._yOff=smoothedStem.get(c._id);
  }

  const visibleIds=new Set();
  for(const c of visible){
    visibleIds.add(c._id);
    const pinTop=GROUND_Y-c._yOff;
    if(!pinEls.has(c._id)){
      const el=document.createElement('div');
      el.className='pin'+(c.isGrave?' is-grave':'');
      el.innerHTML=`<div class="pin-bubble" dir="${isHe?'rtl':'ltr'}">
        <div class="pin-name ${isHe?'serif-he':'serif-la'}">${c.name}</div>
        ${c.name2?`<div class="pin-name2">${c.name2}</div>`:''}
        <div class="pin-sublabel">${c.sublabel}</div>
        <div class="pin-dist"></div>
        <div class="pin-bear"></div>
      </div>
      <div class="pin-spike"></div>
      <div class="pin-dot"></div>`;
      el.addEventListener('click',()=>openDetail({...c,dist:haversine(S.lat,S.lng,c.lat,c.lng),bear:bearing(S.lat,S.lng,c.lat,c.lng)}));
      overlay.appendChild(el);
      pinEls.set(c._id,el);
    }
    const el=pinEls.get(c._id);
    el.style.left=`${c._x.toFixed(1)}px`;
    el.style.top=`${pinTop.toFixed(1)}px`;
    el.style.opacity=c._op;
    el.querySelector('.pin-dist').textContent=fmtDist(c.dist);
    el.querySelector('.pin-bear').textContent=`${Math.round(c.bear)}° ${cardDir(c.bear)}`;
    el.querySelector('.pin-spike').style.display=c._isBottom?'':'none';
    el.querySelector('.pin-dot').style.display=c._isBottom?'':'none';
  }
  for(const[id,el] of pinEls){
    if(!visibleIds.has(id)){el.remove();pinEls.delete(id);smoothedPos.delete(id);smoothedStem.delete(id);}
  }
}

function renderRadar(){
  const canvas=document.querySelector('#radar-wrap canvas');
  if(!canvas||!S.gpsOK) return;
  const ctx=canvas.getContext('2d');
  const W=canvas.width,R=W/2;
  ctx.clearRect(0,0,W,W);
  // Background
  ctx.beginPath();ctx.arc(R,R,R-1,0,Math.PI*2);
  ctx.fillStyle='rgba(8,8,6,0.78)';ctx.fill();
  ctx.strokeStyle='rgba(232,224,208,0.18)';ctx.lineWidth=1;ctx.stroke();
  // Range rings
  for(const f of[0.33,0.66]){
    ctx.beginPath();ctx.arc(R,R,(R-2)*f,0,Math.PI*2);
    ctx.strokeStyle='rgba(232,224,208,0.08)';ctx.stroke();
  }
  // FOV wedge
  const fovRad=(S.FOV/2)*(Math.PI/180);
  const north=-Math.PI/2;
  ctx.beginPath();ctx.moveTo(R,R);
  ctx.arc(R,R,R-4,north-fovRad,north+fovRad);
  ctx.closePath();ctx.fillStyle='rgba(200,132,26,0.1)';ctx.fill();
  // Heading line
  ctx.beginPath();ctx.moveTo(R,R);
  ctx.lineTo(R+Math.cos(north)*(R-4),R+Math.sin(north)*(R-4));
  ctx.strokeStyle='rgba(200,132,26,0.5)';ctx.lineWidth=1;ctx.stroke();
  // Cemetery dots
  const cems=filteredCems();
  for(const c of cems){
    if(!S.lat) continue;
    const dist=haversine(S.lat,S.lng,c.lat,c.lng);
    if(dist>S.rangeKm*1.1) continue;
    const bear=bearing(S.lat,S.lng,c.lat,c.lng);
    const angle=(bear-S.heading-90)*(Math.PI/180);
    const rNorm=Math.min(dist/S.rangeKm,1)*(R-8)+4;
    const x=R+Math.cos(angle)*rNorm;
    const y=R+Math.sin(angle)*rNorm;
    ctx.beginPath();ctx.arc(x,y,c.isGrave?2.5:3.5,0,Math.PI*2);
    ctx.fillStyle=c.isGrave?'#e8c040':'#c8841a';ctx.fill();
  }
  // Self dot
  ctx.beginPath();ctx.arc(R,R,4,0,Math.PI*2);
  ctx.fillStyle='rgba(232,224,208,0.9)';ctx.fill();
}

// COMPASS CALIBRATION

const headingSamples=[];
let calibShown=false,calibTimer=null;
function checkCalibration(h){
  headingSamples.push(h);
  if(headingSamples.length>30) headingSamples.shift();
  if(headingSamples.length<15) return;
  // Circular mean & variance
  const sinSum=headingSamples.reduce((s,v)=>s+Math.sin(v*Math.PI/180),0);
  const cosSum=headingSamples.reduce((s,v)=>s+Math.cos(v*Math.PI/180),0);
  const R=Math.sqrt(sinSum*sinSum+cosSum*cosSum)/headingSamples.length;
  const variance=1-R; // 0=perfect, 1=random
  const toast=document.getElementById('calib-toast');
  if(variance>0.08&&!calibShown){
    calibShown=true;
    toast.querySelector('p').textContent=t('calibMsg');
    toast.style.display='block';
    calibTimer=setTimeout(()=>{toast.style.display='none';calibShown=false;},8000);
  } else if(variance<0.04&&calibShown){
    clearTimeout(calibTimer);
    toast.style.display='none';calibShown=false;
  }
}

function setupOrientation(){
  let gotAbs=false;
  window.addEventListener('deviceorientationabsolute',e=>{
    if(e.alpha===null) return;
    gotAbs=true;
    S.heading=((360-e.alpha)%360);
    S.headingOK=true;dot('d-cmp','ok');
    checkCalibration(S.heading);
  },true);
  window.addEventListener('deviceorientation',e=>{
    if(gotAbs) return;
    if(e.webkitCompassHeading!=null) S.heading=e.webkitCompassHeading;
    else if(e.alpha!=null) S.heading=((360-e.alpha)%360);
    else return;
    S.headingOK=true;dot('d-cmp','ok');
    checkCalibration(S.heading);
  });
}

// ═══════════════════════════════════════
(function(){
  let startY=0,pulling=false,triggered=false;
  const THRESHOLD=72;
  const ptr=document.getElementById('ptr');
  const icon=document.getElementById('ptr-icon');
  const txt=document.getElementById('ptr-txt');

  function refresh(){
    triggered=true;
    icon.textContent='↻';
    icon.style.animation='spin .8s linear infinite';
    txt.setAttribute('data-i18n','searching');
    txt.textContent=t('searching');
    // Force fresh fetch from Overpass (ignore KEVARIM_DATA & cache)
    const seen=new Set();
    const r=S.rangeKm*1000;
    if(!S.lat||!S.lng){reset();return;}
    async function go(){
      async function overpass(q){const res=await fetch('https://overpass-api.de/api/interpreter',{method:'POST',body:q});if(!res.ok)throw new Error(`HTTP ${res.status}`);const d=await res.json();return d;}
      const qJ=`[out:json][timeout:30];(node["landuse"="cemetery"]["religion"="jewish"](around:${r},${S.lat},${S.lng});way["landuse"="cemetery"]["religion"="jewish"](around:${r},${S.lat},${S.lng});node["amenity"="grave_yard"]["religion"="jewish"](around:${r},${S.lat},${S.lng});way["amenity"="grave_yard"]["religion"="jewish"](around:${r},${S.lat},${S.lng}););out center;`;
      const qA=`[out:json][timeout:30];(node["landuse"="cemetery"](around:${r},${S.lat},${S.lng});way["landuse"="cemetery"](around:${r},${S.lat},${S.lng});node["amenity"="grave_yard"](around:${r},${S.lat},${S.lng});way["amenity"="grave_yard"](around:${r},${S.lat},${S.lng}););out center;`;
      try{
        let d=await overpass(qJ);
        if(!d.elements.length) d=await overpass(qA);
        const fresh=processElements(d.elements,false,seen);
        S.cems=fresh;S.dataOK=true;dot('d-dat','ok');
        // Phase 2 graves
        try{
          const qG=`[out:json][timeout:20];(node["historic"="tomb"]["name"](around:${r},${S.lat},${S.lng});node["historic"="grave"]["name"](around:${r},${S.lat},${S.lng}););out;`;
          const d2=await overpass(qG);
          const graves=processElements(d2.elements,true,seen);
          if(graves.length) S.cems=[...S.cems,...graves];
        }catch(e){}
        document.getElementById('cem-count').textContent=filteredCems().length;
        document.getElementById('cache-icon').textContent='☁';
        clearPins();
      }catch(e){dot('d-dat','err');}
      reset();
    }
    go();
  }

  function reset(){
    triggered=false;pulling=false;
    ptr.style.height='0';
    icon.style.animation='';
    icon.textContent='↓';
    setTimeout(()=>ptr.style.height='0',300);
  }

  document.addEventListener('touchstart',e=>{
    if(document.getElementById('list-view').classList.contains('open')) return;
    if(document.getElementById('detail').classList.contains('open')) return;
    startY=e.touches[0].clientY;pulling=false;triggered=false;
  },{passive:true});

  document.addEventListener('touchmove',e=>{
    if(triggered) return;
    if(document.getElementById('list-view').classList.contains('open')) return;
    if(document.getElementById('detail').classList.contains('open')) return;
    const dy=e.touches[0].clientY-startY;
    if(dy<10) return;
    pulling=true;
    const h=Math.min(dy*0.45,THRESHOLD+10);
    ptr.style.height=h+'px';
    const ratio=Math.min(dy/THRESHOLD,1);
    icon.style.transform=`rotate(${ratio*180}deg)`;
    icon.style.animation='';
    icon.textContent=dy>THRESHOLD?'↺':'↓';
    txt.textContent=dy>THRESHOLD?t('searching'):'';
  },{passive:true});

  document.addEventListener('touchend',e=>{
    if(!pulling||triggered) return;
    const dy=e.changedTouches[0].clientY-startY;
    if(dy>THRESHOLD) refresh();
    else reset();
  },{passive:true});

  // Spin keyframe
  const st=document.createElement('style');
  st.textContent='@keyframes spin{to{transform:rotate(360deg)}}';
  document.head.appendChild(st);
})();

</script>



<div id="ptr"><span id="ptr-txt"></span><span id="ptr-icon">↓</span></div>
</body></html>
